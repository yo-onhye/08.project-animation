  ```jsx
  // setAttribute를 사용하여 data-xxx 세팅이 가능하지만,
  stepElems[i].setAttribute("data-index", i);

  // 기본 속성인 dataset을 사용할 수 있다.
  stepElems[i].dataset.index = i;
  ```
